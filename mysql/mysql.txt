CREATE TABLE xarch ( id INT NOT NULL, reg INT NOT NULL, utime INT NOT NULL, data INT NOT NULL, PRIMARY KEY (id, reg, utime) ) CHARACTER SET utf8 COLLATE utf8_general_ci;
CREATE TABLE xdata ( id INT NOT NULL, reg INT NOT NULL, utime INT NOT NULL, data INT NOT NULL, diap INT NOT NULL, modif FLOAT NOT NULL,  PRIMARY KEY (id, reg) ) CHARACTER SET utf8 COLLATE utf8_general_ci;
